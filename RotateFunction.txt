int maxRotateFunction(int* nums, int numsSize) {
    long long sum=0,F=0;
    for(int i=0;i<numsSize;i++){
        sum+=nums[i];
        F+=(long long)i * nums[i];
    } 
    long long maxF= F;
    for(int k=1;k<numsSize;k++){
        F=F+sum-(long long)numsSize *  nums[numsSize-k];
        if(F>maxF)maxF=F;
    }
    return(int)maxF;
}
