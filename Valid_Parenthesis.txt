class Solution {
    public boolean checkValidString(String s) {
        int open = 0,close= 0;
        for (int i=0;i<s.length();i++) {
            char c = s.charAt(i);
            if (c == '(') {
                open++;
                close++;
            } else if (c == ')') {
                open--;
                close--;
            } else if (c == '*') {
                open--; 
                close++; 
            }
            if (close < 0) {
                return false;
            }
            if (open < 0) {
                open = 0;
            }
        }
        return open==0;
    }
}
